005package org.usfirst.frc330.Beachbot2014Java.commands;
006
007import edu.wpi.first.wpilibj.command.Command;
008import org.usfirst.frc330.Beachbot2014Java.Robot;
009/*
010 * This will drive the robot forwards to a waypoint on the field based on its 
011 * original starting position.
012 */
013
014public class DriveWaypointBackward extends DriveWaypoint {
015
016    public DriveWaypointBackward(double x, double y, double tolerance, double timeout, boolean stopAtEnd) {
017        super(x, y, tolerance, timeout, stopAtEnd);
018    }
019
020    protected void calcXY(double x, double y) {
021        double gyroAngle;
022        
023        super.calcXY(x, y);
024
025        leftDistance = -leftDistance;
026        rightDistance = -rightDistance;
027        
028        gyroAngle = Robot.chassis.getAngle();
029        if (gyroAngle < angle)
030            angle = angle-180;
031        else
032            angle = angle+180;
033//        System.out.println("Backward Angle: " + angle);
034    }
035
036    public Command copy() {
037        return new DriveWaypointBackward(0,0,0,0,true);
038    }
039    
040}
