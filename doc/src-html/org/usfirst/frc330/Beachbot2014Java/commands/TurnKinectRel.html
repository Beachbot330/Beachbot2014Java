006package org.usfirst.frc330.Beachbot2014Java.commands;
007
008import edu.wpi.first.wpilibj.command.AutoSpreadsheetCommand;
009import edu.wpi.first.wpilibj.command.Command;
010import org.usfirst.frc330.Beachbot2014Java.Robot;
011
012/**
013 *
014 * @author Joe-XPS13-W7
015 */
016public class TurnKinectRel extends TurnGyroRel {
017
018    public TurnKinectRel(double angle) {
019        super(angle);
020    }
021
022    public TurnKinectRel(double angle, double tolerance) {
023        super(angle, tolerance);
024    }
025
026    public TurnKinectRel(double angle, double tolerance, double timeout, boolean stopAtEnd) {
027        super(angle, tolerance, timeout, stopAtEnd);
028    }
029
030    double direction;
031    // Called just before this Command runs the first time
032    protected void initialize() {
033        if (Robot.oi.getLeftKinectJoystick().getY() > Robot.oi.getRightKinectJoystick().getY())
034            direction = 1;
035        else
036            direction = -1;
037        angle = angle*direction;
038        super.initialize();
039    }
040    
041    public Command copy() {
042        return new TurnKinectRel(0);
043    }
044}
