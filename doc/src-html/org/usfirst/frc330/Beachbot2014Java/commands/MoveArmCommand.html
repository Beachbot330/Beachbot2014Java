007package org.usfirst.frc330.Beachbot2014Java.commands;
008
009import edu.wpi.first.wpilibj.command.Command;
010import org.usfirst.frc330.Beachbot2014Java.Robot;
011
012/**
013 *
014 * @author Joe-XPS13-W7
015 */
016public abstract class MoveArmCommand extends Command {
017    double setpoint; 
018    boolean started = false;
019    double outputRange = 0;
020    double startPosition = 0;
021    double accelDistance = 0;
022    double decelDistance = 0;
023    double origDecelDistance = 0;
024    double maxSpeed = 0;
025    double minSpeed = 0;
026
027    public MoveArmCommand(double position) {
028        this(position, 1.0, 0.4, 0.4, 1.2);
029    }
030    
031//maxSpeed       _______
032//              /       \
033//             /         \
034//minSpeed    |           \
035//           accel      decel
036//         Distance    Distance
037    
038    public MoveArmCommand(double position, double maxSpeed, double minSpeed, double accelDistance, double decelDistance) {
039        requires(Robot.arm);
040        requires(Robot.wings);
041        setpoint = position;
042        this.maxSpeed = maxSpeed;
043        this.minSpeed = minSpeed;
044        this.maxSpeed = maxSpeed;
045        this.accelDistance = accelDistance;
046        this.decelDistance = decelDistance;
047        this.origDecelDistance = decelDistance;
048    }
049
050    // Called just before this Command runs the first time
051    protected void initialize() {
052        if (!Robot.arm.areWingsSafeToClose(setpoint))
053            Robot.wings.setWingsOpen();
054        started = false;
055        startPosition = Robot.arm.getArmPosition();
056        outputRange = minSpeed;
057        if (setpoint - startPosition < accelDistance+origDecelDistance)
058            decelDistance = setpoint - accelDistance - startPosition;
059        else 
060            decelDistance = origDecelDistance;
061        if (decelDistance <= 0)
062            decelDistance = 0.001;
063        if (accelDistance <= 0)
064            accelDistance = 0.001;
065//        System.out.println("MoveArmCommand Initialize decelDistance= " + decelDistance + " setpoint " + setpoint + " startPosition " + startPosition + " accelDistance " + accelDistance + " origDecelDistance " + origDecelDistance);
066    }
067
068    // Called repeatedly when this Command is scheduled to run
069    final protected void execute() {
070        double x, y;
071        if ((Robot.wings.areWingsOpen() || Robot.arm.areWingsSafeToClose(setpoint)) && !started) {
072                Robot.arm.setArmSetPoint(setpoint);
073                Robot.arm.setPIDOutputRange(minSpeed); 
074                outputRange = minSpeed;
075                Robot.arm.enable();
076                started = true;
077
078//                System.out.println("outputRange: " + outputRange);                
079
080        } else if (started) {
081
082            if (setpoint > startPosition) {
083                if (Robot.arm.getArmPosition() > setpoint) {  //past setpoint
084                    outputRange = minSpeed;
085                } else if (Robot.arm.getArmPosition() <= (startPosition + accelDistance)) { //in accell range
086                    x = (Robot.arm.getArmPosition() - startPosition)/accelDistance;
087                    y = maxSpeed - minSpeed;
088                    outputRange = y*x+minSpeed;
089                } else if (Robot.arm.getArmPosition() >= (setpoint - decelDistance)) { //in decell range
090                    x = (setpoint - Robot.arm.getArmPosition())/decelDistance;
091                    y = maxSpeed - minSpeed;
092                    outputRange = x*y + minSpeed;
093    //                System.out.println("In Decel Condition x= " + x);
094                } else { //in middle range
095                   outputRange = maxSpeed;
096                }   
097            } else {
098                if (Robot.arm.getArmPosition() < setpoint) {  //past setpoint
099                    outputRange = minSpeed;
100                } else if (Robot.arm.getArmPosition() >= (startPosition - accelDistance)) { //in accell range
101                    x = (startPosition -Robot.arm.getArmPosition())/accelDistance;
102                    y = maxSpeed - minSpeed;
103                    outputRange = y*x+minSpeed;
104                } else if (Robot.arm.getArmPosition() <= (setpoint + decelDistance)) { //in decell range
105                    x = (Robot.arm.getArmPosition() - setpoint )/decelDistance;
106                    y = maxSpeed - minSpeed;
107                    outputRange = x*y + minSpeed;
108    //                System.out.println("In Decel Condition x= " + x);
109                } else { //in middle range
110                   outputRange = maxSpeed;
111                }
112            }
113            
114        }
115        System.out.println("ArmPosition: " + Robot.arm.getArmPosition() + " outputRange: " + outputRange);
116        Robot.arm.setPIDOutputRange(outputRange);
117//      System.out.println("outputRange: " + outputRange);
118        
119    }
120
121    // Make this return true when this Command no longer needs to run execute()
122    final protected boolean isFinished() {
123        return Robot.arm.onTarget() && started;
124    }
125
126    // Called once after isFinished returns true
127    final protected void end() {
128        Robot.arm.setPIDOutputRangeDefault();
129        started=false;
130    }
131
132    // Called when another command which requires one or more of the same
133    // subsystems is scheduled to run
134    final protected void interrupted() {
135        end();
136    }
137    
138}
