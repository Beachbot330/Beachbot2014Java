212        if (armCommand < 0) 
213            armCommand = -(armCommand*armCommand);
214        else
215            armCommand = armCommand*armCommand;
216        if (Math.abs(armCommand) > 0.10)
217        {
218            if (armPID.isEnable())
219                armPID.disable();
220            if (Robot.wings.areWingsOpen())
221                    set(armCommand);

223                set(armCommand);
224            else if (this.isArmInFrontSafePoint(0) && armCommand > 0)

226            else if (this.isArmInBackSafePoint(0) && armCommand < 0)
