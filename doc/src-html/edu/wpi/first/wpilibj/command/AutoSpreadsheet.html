085    public AutoSpreadsheet()
086    {
087//        System.out.println("begin of AutoSpreadsheet Constructor");
088        autoChooser = new SendableChooser();
089//        System.out.println(MarsRock.class.getName());
090        autoChooser.addDefault("Mars Rock", new MarsRock());

095

097    {
098        commandTable.put(command.getName().toUpperCase(), command);
099    }

103        commandTable.put(command.getName().toUpperCase(), command);

107    public CommandGroup getSelected() {
108        if (autoChooser.getSelected() instanceof MarsRock)
109        {
110            CommandGroup cg = new CommandGroup();
111            cg.addSequential(new MarsRock());
112            return (CommandGroup)cg;
113        }
114        else
115            return buildScript((String)autoChooser.getSelected());
116    }

118    public void readScripts()
119    {
120        try {
121            file = (FileConnection) Connector.open(filename, Connector.READ);
122            reader = new BufferedReader(new InputStreamReader(file.openInputStream()));
123           
124            String line;
125            String scriptName = null;
126            int comma1, comma2;
127            boolean scriptStarted = false;
128                        
129            while ((line = reader.readLine()) != null)
130            {
131//                System.out.println(line);
132                if (line.toUpperCase().startsWith("SCRIPT_NAME,"))
133                {
134                    comma1 = line.indexOf(",");
135                    comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
136                    scriptName = line.substring(comma1+1,comma2).toUpperCase().trim();
137                    System.out.println("Found Script: " + scriptName);
138                    scriptStarted = true;
139                    autoChooser.addObject(scriptName, scriptName);
140                }
141                else if (line.toUpperCase().startsWith("END_OF_SPREAD_SHEET,"))
142                {
143//                        System.out.println("End of Spreadsheet");
144                        break;                     
145                }
146            }           
147            SmartDashboard.putData("Autonomous mode", autoChooser);
148            
149        } catch (IOException ex) {
150            ex.printStackTrace();
151        }
152        finally
153        {
154            try {
155                file.close();
156            } catch (IOException ex) {
157                ex.printStackTrace();
158            }
159        }
160    }

162    public CommandGroup buildScript(String scriptToRead)
163    {
164        CommandGroup cg = null;
165        try {
166            file = (FileConnection) Connector.open(filename, Connector.READ);
167            reader = new BufferedReader(new InputStreamReader(file.openInputStream()));
168           
169            String line;
170            String scriptName = null;
171            int comma1, comma2;
172            boolean scriptStarted = false;
173            String commandName = null;
174            double timeout, param1, param2, param3;
175            boolean stop;
176            boolean sequential;
177            
178
179            Command command = null;
180              
181            //search for specified script name
182            while ((line = reader.readLine()) != null)
183            {
184//                System.out.println(line);
185                if (line.toUpperCase().startsWith("SCRIPT_NAME,"))
186                {
187                    comma1 = line.indexOf(",");
188                    comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
189                    scriptName = line.substring(comma1+1,comma2).toUpperCase().trim();
190                    System.out.println("Found Script: " + scriptName);
191                    scriptStarted = true;
192                    if (scriptName.equals(scriptToRead))
193                    {
194                        cg = new CommandGroup(scriptName);
195                        break;
196                    }
197                }
198            }
199            
200            while ((line = reader.readLine()) != null)
201            {
202                if (line.toUpperCase().startsWith("END_OF_SPREAD_SHEET,"))
203                {
204                    if (scriptStarted)
205                        System.err.println("!!!End of Spreadsheet found while " + scriptName + " still open");
206                    else
207                    {
208//                        System.out.println("End of Spreadsheet");
209                        break;
210                    }
211                }
212                else if (line.toUpperCase().startsWith("END,"))
213                {
214//                    System.out.println("End of Script: " + scriptName);
215                    scriptStarted = false;
216                    cg.addSequential(new Wait(15));
217                    break;
218                }
219                else if (line.startsWith(",")|| line.length() <= 2 )
220                {
221                    //empty line
222                }
223                else
224                {
225                    if (!scriptStarted)
226                        System.err.println("!!!Command found while Script not open");
227                    else
228                    {
229                        commandName = line.substring(0, line.indexOf(","));
230                        
231                        comma1 = line.indexOf(",");
232                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
233                        if (line.substring(comma1+1).startsWith("S"))
234                            sequential = true;
235                        else
236                            sequential = false;
237                        
238                        line = line.substring(comma1+1);
239                        comma1 = line.indexOf(",");
240                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
241                        try
242                        {
243                            timeout = Double.parseDouble(line.substring(comma1+1,comma2).toUpperCase().trim());
244                        }
245                        catch (NumberFormatException ex)
246                        {
247                            timeout = 0;
248                        }
249
250                        line = line.substring(comma2+1);
251                        if (line.toUpperCase().startsWith("CONTINUE"))
252                            stop = false;
253                        else
254                            stop = true;
255//                        System.out.println("stop: " + stop + " " + line.toUpperCase());
256                        
257                        comma1 = line.indexOf(",");
258                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
259                        try
260                        {
261                            param1 = Double.parseDouble(line.substring(comma1+1,comma2).trim());  
262                        }
263                        catch (NumberFormatException ex)
264                        {
265                            param1 = 0;
266                        }
267                        
268                        comma1 = comma2;
269                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
270                        try
271                        {
272                            param2 = Double.parseDouble(line.substring(comma1+1,comma2).trim());  
273                        }
274                        catch (NumberFormatException ex)
275                        {
276                            param2 = 0;
277                        }
278                        
279                        comma1 = comma2;
280                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
281                        try
282                        {
283                            param3 = Double.parseDouble(line.substring(comma1+1,comma2).trim());  
284                        }
285                        catch (NumberFormatException ex)
286                        {
287                            param3 = 0;
288                        }                       
289                        
290                        command = (Command)commandTable.get(commandName.toUpperCase());
291                       
292                        
293                        if (command == null)
294                        {
295                            System.err.println("Could not find command: " + commandName);
296                        }
297                        else if (command instanceof AutoSpreadsheetCommand)
298                        {
299                            System.out.println("Found Command: " + command.getName());
300                            command = ((AutoSpreadsheetCommand)command).copy();
301                            ((AutoSpreadsheetCommand)command).setStopAtEnd(stop);
302                            ((AutoSpreadsheetCommand)command).setParam1(param1);
303                            ((AutoSpreadsheetCommand)command).setParam2(param2);
304                            ((AutoSpreadsheetCommand)command).setParam3(param3);
305                            (command).setTimeout(timeout);
306                            if (sequential)   
307                                cg.addSequential(command);
308                            else
309                                cg.addParallel(command);
310                            
311                        }
312                        else if (command instanceof AutoSpreadsheetCommandGroup)
313                        {
314                            System.out.println("Found Command Group: " + command.getName());
315                            command = ((AutoSpreadsheetCommandGroup)command).copy();
316                            if (sequential)   
317                                cg.addSequential(command);
318                            else
319                                cg.addParallel(command);
320                        }
321                        else
322                        {
323                            System.err.println(commandName + " (" + command.getName() + ") is not instance of AutoSpreadsheetCommand");
324                        }
325
326                        
327//                        System.out.println("Command: " + commandName + " Timeout: " + timeout + " Continue: " + !stop + " Param1: " + param1 + " Param2: " + param2 + " Param3: " + param3);
328                    }
329                }
330            }
331           
332        } catch (IOException ex) {
333            ex.printStackTrace();
334        }
335        finally
336        {
337            try {
338                file.close();
339            } catch (IOException ex) {
340                ex.printStackTrace();
341            }
342        }
343        return cg;
344    }
