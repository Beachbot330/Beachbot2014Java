064public class AutoSpreadsheet {
065    FileConnection file = null;
066    BufferedReader reader = null;
067    final String filename = "file:///2013AutoModesJava.csv";
068    SendableChooser autoChooser;
069    
070    public AutoSpreadsheet()
071    {
072//        System.out.println("begin of AutoSpreadsheet Constructor");
073        autoChooser = new SendableChooser();
074//        System.out.println(MarsRock.class.getName());
075        autoChooser.addDefault("Mars Rock", new MarsRock());
076//        System.out.println("end of AutoSpreadsheet Constructor");
077    }

079    private Hashtable commandTable = new Hashtable();
080    
081    public void addCommand(AutoSpreadsheetCommand command)
082    {
083        commandTable.put(command.getName().toUpperCase(), command);
084    }
085
086    public void addCommand(AutoSpreadsheetCommandGroup command)
087    {
088        commandTable.put(command.getName().toUpperCase(), command);
089    }

092    public CommandGroup getSelected() {
093        if (autoChooser.getSelected() instanceof MarsRock)
094        {
095            CommandGroup cg = new CommandGroup();
096            cg.addSequential((Command)autoChooser.getSelected());
097            return (CommandGroup)cg;
098        }
099        else
100            return buildScript((String)autoChooser.getSelected());
101    }

103    public void readScripts()
104    {
105        try {
106            file = (FileConnection) Connector.open(filename, Connector.READ);
107            reader = new BufferedReader(new InputStreamReader(file.openInputStream()));
108           
109            String line;
110            String scriptName = null;
111            int comma1, comma2;
112            boolean scriptStarted = false;
113                        
114            while ((line = reader.readLine()) != null)
115            {
116//                System.out.println(line);
117                if (line.toUpperCase().startsWith("SCRIPT_NAME,"))
118                {
119                    comma1 = line.indexOf(",");
120                    comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
121                    scriptName = line.substring(comma1+1,comma2).toUpperCase().trim();
122//                    System.out.println("Found Script: " + scriptName);
123                    scriptStarted = true;
124                    autoChooser.addObject(scriptName, scriptName);
125                }
126                else if (line.toUpperCase().startsWith("END_OF_SPREAD_SHEET,"))
127                {
128//                        System.out.println("End of Spreadsheet");
129                        break;                     
130                }
131            }           
132            SmartDashboard.putData("Autonomous mode", autoChooser);
133            
134        } catch (IOException ex) {
135            ex.printStackTrace();
136        }
137        finally
138        {
139            try {
140                file.close();
141            } catch (IOException ex) {
142                ex.printStackTrace();
143            }
144        }
145    }

147    public CommandGroup buildScript(String scriptToRead)
148    {
149        CommandGroup cg = null;
150        try {
151            file = (FileConnection) Connector.open(filename, Connector.READ);
152            reader = new BufferedReader(new InputStreamReader(file.openInputStream()));
153           
154            String line;
155            String scriptName = null;
156            int comma1, comma2;
157            boolean scriptStarted = false;
158            String commandName = null;
159            double timeout, param1, param2, param3;
160            boolean stop;
161            boolean sequential;
162            
163
164            Command command = null;
165              
166            //search for specified script name
167            while ((line = reader.readLine()) != null)
168            {
169//                System.out.println(line);
170                if (line.toUpperCase().startsWith("SCRIPT_NAME,"))
171                {
172                    comma1 = line.indexOf(",");
173                    comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
174                    scriptName = line.substring(comma1+1,comma2).toUpperCase().trim();
175//                    System.out.println("Found Script: " + scriptName);
176                    scriptStarted = true;
177                    if (scriptName.equals(scriptToRead))
178                    {
179                        cg = new CommandGroup(scriptName);
180                        break;
181                    }
182                }
183            }
184            
185            while ((line = reader.readLine()) != null)
186            {
187                if (line.toUpperCase().startsWith("END_OF_SPREAD_SHEET,"))
188                {
189                    if (scriptStarted)
190                        System.err.println("!!!End of Spreadsheet found while " + scriptName + " still open");
191                    else
192                    {
193//                        System.out.println("End of Spreadsheet");
194                        break;
195                    }
196                }
197                else if (line.toUpperCase().startsWith("END,"))
198                {
199//                    System.out.println("End of Script: " + scriptName);
200                    scriptStarted = false;
201                    break;
202                }
203                else if (line.startsWith(",")|| line.length() <= 2 )
204                {
205                    //empty line
206                }
207                else
208                {
209                    if (!scriptStarted)
210                        System.err.println("!!!Command found while Script not open");
211                    else
212                    {
213                        commandName = line.substring(0, line.indexOf(","));
214                        
215                        comma1 = line.indexOf(",");
216                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
217                        if (line.substring(comma1+1).startsWith("S"))
218                            sequential = true;
219                        else
220                            sequential = false;
221                        
222                        line = line.substring(comma1+1);
223                        comma1 = line.indexOf(",");
224                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
225                        try
226                        {
227                            timeout = Double.parseDouble(line.substring(comma1+1,comma2).toUpperCase().trim());
228                        }
229                        catch (NumberFormatException ex)
230                        {
231                            timeout = 0;
232                        }
233
234                        line = line.substring(comma2+1);
235                        if (line.toUpperCase().startsWith("CONTINUE"))
236                            stop = false;
237                        else
238                            stop = true;
239                        
240                        comma1 = line.indexOf(",");
241                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
242                        try
243                        {
244                            param1 = Double.parseDouble(line.substring(comma1+1,comma2).trim());  
245                        }
246                        catch (NumberFormatException ex)
247                        {
248                            param1 = 0;
249                        }
250                        
251                        comma1 = comma2;
252                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
253                        try
254                        {
255                            param2 = Double.parseDouble(line.substring(comma1+1,comma2).trim());  
256                        }
257                        catch (NumberFormatException ex)
258                        {
259                            param2 = 0;
260                        }
261                        
262                        comma1 = comma2;
263                        comma2 = line.substring(comma1+1).indexOf(",")+comma1+1;
264                        try
265                        {
266                            param3 = Double.parseDouble(line.substring(comma1+1,comma2).trim());  
267                        }
268                        catch (NumberFormatException ex)
269                        {
270                            param3 = 0;
271                        }                       
272                        
273                        command = (Command)commandTable.get(commandName.toUpperCase());
274                       
275                        
276                        if (command == null)
277                        {
278                            System.err.println("Could not find command: " + commandName);
279                        }
280                        else if (command instanceof AutoSpreadsheetCommand)
281                        {
282                            command = ((AutoSpreadsheetCommand)command).copy();
283                            ((AutoSpreadsheetCommand)command).setStopAtEnd(stop);
284                            ((AutoSpreadsheetCommand)command).setParam1(param1);
285                            ((AutoSpreadsheetCommand)command).setParam2(param2);
286                            ((AutoSpreadsheetCommand)command).setParam3(param3);
287                            ((AutoSpreadsheetCommand)command).setTimeout(timeout);
288                            if (sequential)   
289                                cg.addSequential(command);
290                            else
291                                cg.addParallel(command);
292                            
293                        }
294                        else if (command instanceof AutoSpreadsheetCommandGroup)
295                        {
296                            System.out.println("Found Command Group: " + command.getName());
297                            command = ((AutoSpreadsheetCommandGroup)command).copy();
298                            if (sequential)   
299                                cg.addSequential(command);
300                            else
301                                cg.addParallel(command);
302                        }
303                        else
304                        {
305                            System.err.println(commandName + " (" + command.getName() + ") is not instance of AutoSpreadsheetCommand");
306                        }
307
308                        
309//                        System.out.println("Command: " + commandName + " Timeout: " + timeout + " Continue: " + !stop + " Param1: " + param1 + " Param2: " + param2 + " Param3: " + param3);
310                    }
311                }
312            }
313           
314        } catch (IOException ex) {
315            ex.printStackTrace();
316        }
317        finally
318        {
319            try {
320                file.close();
321            } catch (IOException ex) {
322                ex.printStackTrace();
323            }
324        }
325        return cg;
326    }
