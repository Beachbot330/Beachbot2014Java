030public interface AutoSpreadsheetCommand{
031    /**
032     * The first parameter in the AutoSpreadsheet. This should be overridden by
033     * the AutoSpreadSheetCommand to do something useful.
034     * @param param1 
035     */
036    public abstract void setParam1(double param1);
037    /**
038     * The second parameter in the AutoSpreadsheet. This should be overridden by
039     * the AutoSpreadSheetCommand to do something useful.
040     * @param param2 
041     */
042    public abstract void setParam2(double param2);
043    /**
044     * The third parameter in the AutoSpreadsheet. This should be overridden by
045     * the AutoSpreadSheetCommand to do something useful.
046     * @param param3 
047     */
048    public abstract void setParam3(double param3);
049    public abstract void setStopAtEnd(boolean stopAtEnd);
050    /**
051     * 
052     * @return a copy of the command
053     */
054    public abstract Command copy();
055    
056}
